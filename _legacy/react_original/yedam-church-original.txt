        const CustomCursor = () => {
            const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });
            const [isHovering, setIsHovering] = useState(false);
            const [isDarkTheme, setIsDarkTheme] = useState(false);
            const [isTouchDevice, setIsTouchDevice] = useState(false);

            useEffect(() => {
                if ('ontouchstart' in window || navigator.maxTouchPoints > 0) {
                    setIsTouchDevice(true);
                    return; 
                }

                const updatePos = (e) => {
                    setMousePosition({ x: e.clientX, y: e.clientY });
                    const el = document.elementFromPoint(e.clientX, e.clientY);
                    if (el) {
                        const darkSection = el.closest('[data-theme="dark"]');
                        setIsDarkTheme(!!darkSection);
                    }
                };
                const handleHover = (e) => {
                    const target = e.target.closest('a, button, .hover-trigger, .group');
                    setIsHovering(!!target);
                };
                window.addEventListener('mousemove', updatePos);
                window.addEventListener('mouseover', handleHover);
                return () => {
                    window.removeEventListener('mousemove', updatePos);
                    window.removeEventListener('mouseover', handleHover);
                };
            }, []);

            if (isTouchDevice) return null;

            return (
                <>
                    <motion.div
                        className="fixed top-0 left-0 w-96 h-96 rounded-full pointer-events-none z-[1]"
                        animate={{
                            x: mousePosition.x - 192,
                            y: mousePosition.y - 192,
                            opacity: isDarkTheme ? 0.15 : 0,
                        }}
                        transition={{ type: 'tween', ease: 'linear', duration: 0 }}

        const fetchWithBackoff = async (url, options, retries = 5) => {
            const delays = [1000, 2000, 4000, 8000, 16000];
            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    return await response.json();
                } catch (e) {
                    if (i === retries - 1) throw e;
                    await new Promise(res => setTimeout(res, delays[i]));
                }
            }
        };
